// Java 21 Virtual Threads 예시
import java.util.concurrent.*;

public class MiniExecutor {
  private final ExecutorService ex = Executors.newVirtualThreadPerTaskExecutor();
  public <T> Future<T> submit(Callable<T> task){
    return ex.submit(task);
  }
  public void close(){ ex.shutdown(); }
  public static void main(String[] args) throws Exception {
    var pool = new MiniExecutor();
    var f = pool.submit(() -> "Hello, AI" );
    System.out.println("Result: " + f.get());
    pool.close();
  }
}

// C++20 lock-free-ish ring buffer (단순화)
#include <array>
#include <optional>

template <typename T, std::size_t N>
class RingQueue {
  std::array<T,N> buf{}; size_t h=0,t=0; bool full=false;
public:
  bool push(const T& x){ if(full) return false; buf[t]=x; t=(t+1)%N; full=(t==h); return true; }
  std::optional<T> pop(){ if(empty()) return std::nullopt; auto v=buf[h]; h=(h+1)%N; full=false; return v; }
  bool empty() const { return (!full && h==t); }
};
